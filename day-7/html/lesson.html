//Topic 6 Reduce
const numbers = [1, 2, 3, 4, 5];

// Sum
const total = numbers.reduce((acc, n) => acc + n, 0);
console.log(total); 

// Step by step:
// start: acc = 0
// 1st:   acc = 0 + 1 = 1
// 2nd:   acc = 1 + 2 = 3
// 3rd:   acc = 3 + 3 = 6
// 4th:   acc = 6 + 4 = 10
// 5th:   acc = 10 + 5 = 15

// Max value
const max = numbers.reduce((a, n) => n > a ? n : a, -Infinity);
console.log(max);
